rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Permitir acesso a clientes para usuários autenticados
    match /clientes/{document} {
      allow read, write: if request.auth != null;
    }
    
    // Permitir acesso a receitas para usuários autenticados
    match /receitas/{document} {
      allow read, write: if request.auth != null;
    }
    
    // Permitir acesso a despesas para usuários autenticados
    match /despesas/{document} {
      allow read, write: if request.auth != null;
    }
    
    // Permitir acesso a projetos para usuários autenticados
    match /projetos/{document} {
      allow read, write: if request.auth != null;
    }
    
    // Permitir acesso a tarefas para usuários autenticados
    match /tarefas/{document} {
      allow read, write: if request.auth != null;
    }
    
    // Permitir acesso a comentários para usuários autenticados
    match /comentarios/{document} {
      allow read, write: if request.auth != null;
    }
    
    // Permitir acesso a templates de checklist para usuários autenticados
    match /checklist_templates/{document} {
      allow read, write: if request.auth != null;
    }
    
    // Permitir acesso a atividades de projeto para usuários autenticados
    match /atividades_projeto/{document} {
      allow read, write: if request.auth != null;
    }
    
    // Permitir acesso a status personalizados para usuários autenticados
    match /status_personalizados/{document} {
      allow read, write: if request.auth != null;
    }
    
    // Permitir acesso aos perfis de usuário
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Permitir acesso a senhas para usuários autenticados
    match /senhas/{document} {
      allow read, write: if request.auth != null;
    }
    
    // Permitir acesso a atividades dos projetos para usuários autenticados
    match /atividades_projetos/{document} {
      allow read, write: if request.auth != null;
    }
    
    // Permitir acesso a orçamentos para usuários autenticados
    match /orcamentos/{document} {
      allow read, write: if request.auth != null;
    }
    
    // Permitir acesso a recibos para usuários autenticados
    match /recibos/{document} {
      allow read, write: if request.auth != null;
    }
    
    // Permitir leitura/escrita para usuários autenticados (admin)
    match /orders/{document} {
      allow read, write: if request.auth != null;
      // Permitir criação de pedidos para todos (checkout do site)
      allow create: if true;
    }
    
    // Permitir acesso a boards do Kanban para usuários autenticados
    match /kanban_boards/{document} {
      allow read, write: if request.auth != null;
    }
    
    // Permitir acesso a colunas do Kanban para usuários autenticados
    match /kanban_columns/{document} {
      allow read, write: if request.auth != null;
    }
    
    // Permitir acesso a tarefas do Kanban para usuários autenticados
    match /kanban_tasks/{document} {
      allow read, write: if request.auth != null;
    }
  }
}
